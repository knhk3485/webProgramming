#include <stdio.h>
#include <string.h>
#include <sys/types.h>
#include <unistd.h>
#include <stdlib.h>
#include <sys/wait.h>
#include <fcntl.h>
#include <stdbool.h>
#define MAXARG 10
#define MAXCOM 5
 
void check_ioredirection(char *args[], int *ofd, int *ifd);
void check_pipe(char *args[], char *pre_args[], char *post_args[], bool *dopipe);
 
int main()
{
    char buf[256];
    char *coms[MAXCOM];
    char *args[MAXCOM][MAXARG];
    char *pre_args[MAXARG], *post_args[MAXARG];
    char *s;
    char *save;
    int comn, argn;
    static const char delim[] = " \t\n";
    int pid, pid2, status;
    int pfd[2], ofd, ifd;
    bool dopipe;
 
    while (1)
    {
        printf("[shell] ");
        fgets(buf, sizeof(buf), stdin);
        comn = 0;
 
        s = strtok_r(buf, ";", &save);
        while (s)
        {
            coms[comn++] = s;
            s = strtok_r(NULL, ";", &save);
        }
 
        for (int i = 0; i < comn; i++)
        {
            argn = 0;
 
            s = strtok_r(coms[i], delim, &save);
            while (s)
            {
                args[i][argn++] = s;
                s = strtok_r(NULL, delim, &save);
            }
 
            args[i][argn] = (char *)0;
        }
 
        if (!strcmp(args[0][0], "quit"))
            break;
 
        // 명령어 순차적 실행
        for (int i = 0; i < comn; i++)
        {
            if ((pid = fork()) == -1)
            {
                perror("fork failed");
                break;
            }
            else if (pid != 0)
            {
                pid = wait(&status);
            }
            else
            {
                check_pipe(args[i], pre_args, post_args, &dopipe);
 
                if (dopipe == true)
                {
                    if (pipe(pfd) == -1)
                    {
                        perror("pipe error");
                        exit(1);
                    }
 
                    if ((pid2 = fork()) == -1)
                    {
                        perror("fork failed(pipe)");
                        exit(1);
                    }
                    // 파이프에 출력
                    else if (pid2 != 0)
                    {
                        close(pfd[0]);
                        dup2(pfd[1], 1);
                        close(pfd[1]);
                        check_ioredirection(pre_args, &ofd, &ifd);
                        execvp(pre_args[0], pre_args);
                        perror("exec failed(pipe1)");
                        exit(1);
                    }
                    // 파이프로부터 입력
                    else
                    {
                        close(pfd[1]);
                        dup2(pfd[0], 0);
                        close(pfd[0]);
                        check_ioredirection(post_args, &ofd, &ifd);
                        execvp(post_args[0], post_args);
                        perror("exec failed(pipe2)");
                        exit(1);
                    }
                }
                else
                {
                    check_ioredirection(args[i], &ofd, &ifd);
                    execvp(args[i][0], args[i]);
                    perror("exec failed");
                    exit(1);
                }
            }
        }
    }
    exit(0);
}
 
void check_ioredirection(char *args[], int *ofd, int *ifd)
{
    for (int i = 0; args[i + 1] != (char *)0; i++)
    {
        // 출력 재지정
        if (strcmp(args[i], ">") == 0)
        {
            *ofd = open(args[i + 1], O_WRONLY | O_CREAT | O_TRUNC, 0600);
            dup2(*ofd, 1);
            close(*ofd);
            int j;
            for (j = i; args[j + 1] != (char *)0; j++)
                args[j] = args[j + 2];
            args[j] = (char *)0;
            i--;
        }
        // 입력 재지정
        else if (strcmp(args[i], "<") == 0)
        {
            *ifd = open(args[i + 1], O_RDONLY);
            dup2(*ifd, 0);
            close(*ifd);
            int j;
            for (j = i; args[j + 1] != (char *)0; j++)
                args[j] = args[j + 2];
            args[j] = (char *)0;
            i--;
        }
    }
}
 
void check_pipe(char *args[], char *pre_args[], char *post_args[], bool *dopipe)
{
    *dopipe = false;
    int i, j;
    for (i = 0; args[i] != (char *)0; i++)
    {
        if (strcmp(args[i], "|") == 0)
        {
            j = i + 1;
            pre_args[i] = (char *)0;
            *dopipe = true;
            continue;
        }
 
        if (*dopipe == false)
            pre_args[i] = args[i];
        else
            post_args[i - j] = args[i];
    }
    post_args[i - j] = (char *)0;
}
